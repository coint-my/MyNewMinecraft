#version 430 core
layout (location = 0) in vec3 aPos; // Позиция вершины куба из VBO

struct MyInstanceDataRender 
{
    int x_y;
    int z_texIndex;
    int visible_index;
    int index;
};

struct InstanceData 
{
    vec3 pos;
    int texIndex;
    int isVisible;
    int index;
};

InstanceData unpaked(MyInstanceDataRender _instance)
{
    InstanceData myData;
    //vec3 pos = vec3(float((_instance.x_y << 16) >> 16), float(_instance.x_y >> 16), float((_instance.z_texIndex << 16) >> 16));
    vec3 pos = vec3(float(_instance.x_y >> 16), float((_instance.x_y << 16) >> 16), float(_instance.z_texIndex >> 16));
    myData.pos = pos;
    myData.texIndex = (_instance.z_texIndex << 16) >> 16;
    myData.isVisible = (_instance.visible_index << 16) >> 16;
    myData.index = _instance.index;
    return myData;
}

// Тот же буфер, что и в основном рендере (binding = 0)
