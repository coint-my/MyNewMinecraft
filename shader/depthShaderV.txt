#version 430 core
layout (location = 0) in vec3 aPos; // Позиция вершины куба из VBO

struct InstanceData 
{
    mat4 model;
    int texIndex;
    int isVisible;
    int index;
    float p1;
};

// Тот же буфер, что и в основном рендере (binding = 0)
layout (std430, binding = 0) buffer InBuf 
{
    InstanceData allInstances[];
};

uniform mat4 lightSpaceMatrix;

void main()
{
    // Получаем матрицу модели конкретного видимого куба
    if(allInstances[gl_InstanceID].isVisible == 1)
    {
        mat4 model = allInstances[gl_InstanceID].model;
        gl_Position = lightSpaceMatrix * model * vec4(aPos, 1.0);
	return;
    }

    gl_Position = vec4(2.0, 2.0, 2.0, 1.0); // Выбрасываем за экран
}
