#version 430 core
layout (location = 0) in vec3 aPos; // Позиция вершины куба из VBO

struct InstanceData { mat4 model; int texIndex; float p1, p2, p3; };

// Тот же буфер, что и в основном рендере (binding = 1)
layout (std430, binding = 1) buffer OutBuf { InstanceData visibleInstances[]; };

uniform mat4 lightSpaceMatrix;

void main()
{
    // Получаем матрицу модели конкретного видимого куба
    mat4 model = visibleInstances[gl_InstanceID].model;
   
    gl_Position = lightSpaceMatrix * model * vec4(aPos, 1.0);
}
