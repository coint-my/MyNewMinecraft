#version 430 core
layout(local_size_x = 64) in; // Группы по 64 потока

struct InstanceData { mat4 model; int texIndex; int isVisible; };

layout(std430, binding = 0) buffer InBuf { InstanceData allInstances[]; };
layout(std430, binding = 1) buffer OutBuf { InstanceData visibleInstances[]; };
layout(std430, binding = 2) buffer CmdBuf 
{
    uint count;
    uint instanceCount; // Атомарно увеличиваем
    uint firstIndex;
    uint baseVertex;
    uint baseInstance;
} cmd;

uniform mat4 viewProj;

void main() 
{
    uint idx = gl_GlobalInvocationID.x;
    if (idx >= allInstances.length()) return;

    mat4 model = allInstances[idx].model;
    vec4 pos = model[3]; // Позиция куба (4-й столбец)

    // Упрощенный Frustum Culling (проверка, что объект в пределах экрана)
    vec4 clipPos = viewProj * pos;
    float w = clipPos.w + 0.5; // +радиус сферы куба для запаса
    bool visible = (clipPos.x > -w && clipPos.x < w &&
                    clipPos.y > -w && clipPos.y < w &&
                    clipPos.z > 0 && clipPos.z < w);

    if (visible && allInstances[idx].isVisible == 1)
    {
        uint outIdx = atomicAdd(cmd.instanceCount, 1);
        visibleInstances[outIdx] = allInstances[idx];
    }
}
